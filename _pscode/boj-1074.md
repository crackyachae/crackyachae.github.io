def Z(N, r, c):
    if N == 0:
        return 0
    unit = 2 ** (N - 1)
    return (unit ** 2) * ((r // unit) * 2 + c // unit) + Z(N - 1, r % unit, c % unit)


N, r, c = map(int, input().split())
print(Z(N, r, c))
