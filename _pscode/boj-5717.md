```python
while True:
    F, M = map(int, input().split())
    # 입력받은 두 수가 모두 0인 경우에만
    if F == 0 and M == 0:
        # `break`로 종료한다.
        break
    # 그 외는
    else:
        # 두 수의 합을 출력한다.
        print(F + M)
```
