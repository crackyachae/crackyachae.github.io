```python
def get_reward(num):
    a, b, c = num
    return ([c, 10 + b, 100 + 10 * b][num.count(b) - 1]) * 100


N = int(input())
r = []

for _ in range(N):
    m = get_reward(sorted(map(int, input().split())))
    r.append(m)

print(max(r))
```

## 참고 답안

```python
from sys import stdin

t = int(stdin.readline().strip())
res = -1
for _ in range(t):
    a, b, c = map(int, stdin.readline().split())
    if a == b and b == c:
        num = 10000 + a * 1000
    elif a == b or a == c:
        num = 1000 + a * 100
    elif b == c:
        num = 1000 + b * 100
    else:
        p = max(a, max(b, c))
        num = p * 100
    res = max(res, num)
print(res)
```
